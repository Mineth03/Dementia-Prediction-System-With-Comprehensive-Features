app = Flask(__name__)

# Connect to MySQL
#to be connected
def get_db_connection():
    return mysql.connector.connect(
        host="localhost",
        user="root",
        password="",
        database="dementia_tracking"
    )

@app.route('/register-patient', methods=['POST'])
def register_patient():
    data = request.json
    conn = get_db_connection()
    cursor = conn.cursor()

    # Insert patient information
    cursor.execute("""
        INSERT INTO patients (name, email, phone, age, stage_of_dementia, precautions)
        VALUES (%s, %s, %s, %s, %s, %s)
    """, (data['name'], data['email'], data['phone'], data['age'], data['stage_of_dementia'], data['precautions']))

    patient_id = cursor.lastrowid

    # Insert medications
    for medication in data['medications']:
        cursor.execute("""
            INSERT INTO medications (patient_id, medication_name, dosage, scheduled_time)
            VALUES (%s, %s, %s, %s)
        """, (patient_id, medication['medication_name'], medication['dosage'], medication['scheduled_time']))

    # Insert daily routine
    for routine in data['daily_routine']:
        cursor.execute("""
            INSERT INTO daily_routine (patient_id, routine_name, scheduled_time)
            VALUES (%s, %s, %s)
        """, (patient_id, routine['routine_name'], routine['scheduled_time']))

    conn.commit()
    cursor.close()
    conn.close()

    return jsonify({"message": "Patient registered successfully"}), 201

if __name__ == '__main__':
    app.run(debug=True)
