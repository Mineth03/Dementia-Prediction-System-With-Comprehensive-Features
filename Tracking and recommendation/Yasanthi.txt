<script>
    // JavaScript to submit the form data to the backend
    function submitForm() {
        const patientData = {
            name: document.getElementById('name').value,
            age: document.getElementById('age').value,
            stage_of_dementia: document.getElementById('stage').value,
            medications: [
                document.getElementById('donepezil').checked ? 'Donepezil' : '',
                document.getElementById('memantine').checked ? 'Memantine' : ''
            ].filter(Boolean),
            daily_routine: {
                morning_exercise_time: document.getElementById('exercise_time').value,
                brain_activity_time: document.getElementById('brain_activity_time').value,
                sleep_time: document.getElementById('sleep_time').value
            }
        };

        // Send the data to the backend using Fetch API
        fetch('http://localhost:5000/register-patient', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify(patientData)
        })
        .then(response => response.json())
        .then(data => {
            alert('Patient registered successfully');
            console.log(data);
        })
        .catch((error) => {
            console.error('Error:', error);
        });
    }
</script>
